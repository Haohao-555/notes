  function countDown(d) {
        // 1、设置开始的时间（月份是从 0 开始计算的）
        // var startTime = new Date(2021, 7, 2, 1, 15, 0);
        // var startTime = new Date("2021-8-2 18:15");
        var startTime = new Date();
        // 2、把活动截止时间转成日期
        var endTime = new Date(d);
        // 3、两个时间对象相减，不用转成时间戳，会自动转化
        var countTime = endTime - startTime;
        // 4、把时间戳（毫秒）转成 时 分 秒
        var seconds = parseInt(countTime / 1000); // 总秒数
        // 剩余天数
        var day = parseInt(seconds / 60 / 60 / 24);
        // 剩余小时数
        var hours = parseInt(seconds / 3600) % 24;
        // 剩余分钟数
        var mins = parseInt(seconds / 60) % 60;
        // 剩余秒数
        var sec = seconds % 60;

        return day + "天" + hours + "时" + mins + "分" + sec + "秒";
      }
      setInterval(function () {
        /*
          要传递的参数应该是具体活动结束的时间
           2021-8-5 18:15:0 转成的月份是 8 月 不是 9 月
        */
        console.log(countDown("2021-8-5 18:15:0"));
      }, 1000); //反复执行函数本身
